* Peano Arithmetic Sum
  - Recursively Decrement Num1, Increment Num2.
  - Until Num1 == 0, Num2 == Num1 + Num2

#+BEGIN_SRC scheme
(define (peano-sum x y)
  (define (dec x) (- x 1))
  (define (inc x) (+ x 1))
  (if (= x 0)
      y
      (peano-sum (dec x) (inc y))))

(peano-sum 22 11)
#+END_SRC
